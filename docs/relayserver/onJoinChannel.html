onJoinChannel (RelayServer)

C++
bool onJoinChannel (Lacewing::RelayServer &Server, Lacewing::RelayServer::Client &Client
    Lacewing::RelayServer::Channel &Channel)
{
    return true;
}

Server.onJoinChannel (onJoinChannel); /* to register the handler */
C
lw_bool on_join_channel (lw_relaysrv * server, lw_relaysrv_client * client,
    lw_relaysrv_channel * channel)
{
    return lw_true;
}

lw_relaysrv_onjoinchannel (server, on_join_channel); /* to register the handler */
JS
The RelayServer class is not currently supported in Javascript

Description

The onJoinChannel handler is called when a Client attempts to join a channel.

onSetName will always be called before this handler, and this handler will never be called after onDisconnect.

Return true to permit the client to join the channel, or false to prevent it from joining.
